<script>
const canvas = document.getElementById("game");
const ctx = canvas.getContext("2d");

let player, target, obstacles;

function startGame(){
player = {x:140,y:360,size:20};
target = {x:140,y:20,size:25};
obstacles = [
{x:50,y:200,w:200,h:15},
{x:0,y:120,w:180,h:15},
{x:120,y:280,w:180,h:15}
];
document.getElementById("msg").style.display="none";
draw();
}
startGame();

function draw(){
ctx.clearRect(0,0,300,400);

// target â¤ï¸
ctx.fillStyle="red";
ctx.beginPath();
ctx.arc(target.x,target.y,target.size,0,Math.PI*2);
ctx.fill();

// player ğŸ’—
ctx.fillStyle="pink";
ctx.beginPath();
ctx.arc(player.x,player.y,player.size,0,Math.PI*2);
ctx.fill();

// obstacles ğŸ’”
ctx.fillStyle="black";
obstacles.forEach(o=>{
ctx.fillRect(o.x,o.y,o.w,o.h);
});

requestAnimationFrame(draw);
}

// touch move
canvas.addEventListener("touchmove", e=>{
let rect = canvas.getBoundingClientRect();
let touch = e.touches[0];
player.x = touch.clientX - rect.left;
player.y = touch.clientY - rect.top;
checkCollision();
});

// mouse move (for laptop)
canvas.addEventListener("mousemove", e=>{
let rect = canvas.getBoundingClientRect();
player.x = e.clientX - rect.left;
player.y = e.clientY - rect.top;
checkCollision();
});

function checkCollision(){
// win
let dx = player.x-target.x;
let dy = player.y-target.y;
let dist = Math.sqrt(dx*dx+dy*dy);
if(dist < player.size + target.size){
document.getElementById("msg").style.display="block";
}

// hit obstacle
obstacles.forEach(o=>{
if(player.x > o.x && player.x < o.x+o.w &&
player.y > o.y && player.y < o.y+o.h){
startGame();
}
});
}
